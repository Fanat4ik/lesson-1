package ru.geekbrains.DZ_1;

public class DZ_4 {
    //4. Написать метод, принимающий на вход два целых числа и проверяющий,
    // что их сумма лежит в пределах от 10 до 20 (включительно),
    // если да – вернуть true, в противном случае – false.
    public static boolean pruf (int a, int b){   // создали наш метод, который будет получать на вход 2 целых числа, поскольку тип int
        boolean res = false;    // инициализируем и закладываем изначальное значение, оно будет изменяться только при выполнении условий if
        if((a + b) >= 10 && (a + b) <=20) res = true; // проверяем нашу сумму на принадлежность промежутку, используя логическое И
        return res; // возвращаем
    }
    public static void main(String[] args) {
        System.out.println(pruf(5,10)); // выводим результат нашего метода в консоль
    }
}
